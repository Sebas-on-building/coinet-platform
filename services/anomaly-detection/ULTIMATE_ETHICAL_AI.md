## 🎯 Usage Examples

### Complete Ethical AI Pipeline

```typescript
import {
  EthicalAIFramework,
  RealTimeFairnessMonitor,
  AIEthicsCertificationSystem,
  DiversityInclusionFramework,
  IBMAIFairness360Integration
} from '@coinet-ai/anomaly-detection';

// 1. Initialize full ethical system
const ethical = new EthicalAIFramework(config);
const monitor = new RealTimeFairnessMonitor();
const certSystem = new AIEthicsCertificationSystem();
const diversity = new DiversityInclusionFramework();
const aif360 = new IBMAIFairness360Integration();

// 2. Start real-time monitoring
await ethical.start();
monitor.start(60000); // Every minute

// 3. Monitor fairness continuously
monitor.on('fairness_alerts', (alerts) => {
  alerts.forEach(alert => {
    console.log(`⚠️  Fairness issue: ${alert.metric}`);
    console.log(`Auto-correction: ${alert.autoCorrection.applied ? 'Applied' : 'Pending'}`);
  });
});

// 4. Track team diversity
diversity.addTeamMember({
  id: 'dev_001',
  role: 'ML Engineer',
  demographics: { gender: 'female', ethnicity: 'asian' },
  expertise: ['ml', 'python', 'statistics']
});

const divMetrics = diversity.calculateDiversityMetrics();
console.log(`Diversity Score: ${divMetrics.overallDiversityScore}/100`);

// 5. Calculate comprehensive AIF360 metrics
const aifMetrics = await aif360.calculateAIF360Metrics(predictions, 'user_region');
console.log(`Consistency: ${aifMetrics.consistencyScore}`);
console.log(`Theil Index: ${aifMetrics.theilIndex}`);

// 6. Get algorithm recommendations
const algos = await aif360.recommendMitigationAlgorithm(aifMetrics, 'equalized_odds');
console.log(`Recommended: ${algos.map(a => a.name).join(', ')}`);

// 7. Certify model
const certificate = await certSystem.certifyModel(
  'anomaly-detector',
  '3.0.0',
  {
    fairness: biasMetrics,
    explainabilityConfidence: 0.89,
    explainabilityMethods: ['LIME', 'SHAP', 'Counterfactual'],
    gdprCompliant: true,
    auditTrail: true,
    humanOversight: true,
    accuracy: 0.92
  }
);

console.log(`Certificate: ${certificate.level} (${certificate.score}/100)`);
console.log(`Verification: ${certificate.verificationHash}`);

// 8. Generate live dashboard
const dashboard = monitor.getDashboardData();
console.log(`Current Health: ${dashboard.healthScore}/100`);
console.log(`Active Alerts: ${dashboard.alerts.length}`);
```
