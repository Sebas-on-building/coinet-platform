# Copy and paste these commands into your Codespace terminal:

# Navigate to service directory
cd /workspaces/coinet-platform/services/alchemy-whales

# Run the fix script
chmod +x FIX_CODESPACE.sh
./FIX_CODESPACE.sh

# OR manually run these commands:

# 1. Create .env.example (if missing)
cat > .env.example << 'EOF'
ALCHEMY_API_KEY_ETH=your_ethereum_api_key_here
ALCHEMY_API_KEY_POLYGON=your_polygon_api_key_here
ALCHEMY_API_KEY_ARBITRUM=your_arbitrum_api_key_here
ALCHEMY_API_KEY_OPTIMISM=your_optimism_api_key_here
ALCHEMY_API_KEY_BASE=your_base_api_key_here
DATABASE_PASSWORD=your_password_here
WEBHOOK_SECRET=your_webhook_secret_here
DATABASE_HOST=localhost
DATABASE_PORT=5432
DATABASE_NAME=coinet_whales
DATABASE_USER=postgres
REDIS_HOST=localhost
REDIS_PORT=6379
WEBHOOK_PORT=3001
METRICS_PORT=9090
HEALTH_CHECK_PORT=8080
LOG_LEVEL=info
NODE_ENV=development
EOF

# 2. Install dependencies (standalone)
rm -rf node_modules package-lock.json
npm install --legacy-peer-deps

# 3. Build
npm run build

# 4. Create .env
cp .env.example .env

# 5. Edit .env with your keys
code .env

# 6. Start service
npm start

# 7. Verify (in another terminal)
curl http://localhost:8080/health
