# 🚀 Commands to Run in Codespace to Commit All Files

# 1. Navigate to repository root
cd /workspaces/coinet-platform

# 2. Pull latest (to get the ADD_ALL_FILES.sh script)
git pull origin feature/ai-data-feeder

# 3. Run the automated script
chmod +x services/alchemy-whales/ADD_ALL_FILES.sh
./services/alchemy-whales/ADD_ALL_FILES.sh

# 4. Verify what's staged
git status --short | grep "alchemy-whales"

# 5. Commit everything
git commit -m "feat: Complete Alchemy Whales Service - all files"

# 6. Push
git push origin HEAD
