# Copy these commands into your Codespace terminal:

# 1. Pull latest code
cd /workspaces/coinet-platform
git pull origin feature/ai-data-feeder

# 2. Navigate to service
cd services/alchemy-whales

# 3. Install dependencies
npm install

# 4. Build service
npm run build

# 5. Setup environment (choose one method)

# Method A: Create .env file
cp .env.example .env
code .env
# Then edit .env with your API keys

# Method B: Use Codespace secrets (recommended)
# Go to GitHub repo → Settings → Secrets → Codespaces
# Add: ALCHEMY_API_KEY_ETH, ALCHEMY_API_KEY_POLYGON, etc.
# Then restart Codespace

# 6. Start service
npm start

# 7. Verify (in another terminal)
curl http://localhost:8080/health
curl http://localhost:9090/metrics

# Done! Service is running on ports 3001, 8080, 9090
